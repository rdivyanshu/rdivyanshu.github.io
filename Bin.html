Inductive bin : Type :=
| Z
| B0 (n: bin)
| B1 (n: bin).

Fixpoint incr (m: bin) : bin :=
  match m with
  | Z => B1 Z
  | B0 b => B1 b
  | B1 b => B0 (incr b)
  end.

  | B0 b => 2 * (bin_to_nat b)
  | B1 b => 2 * (bin_to_nat b) + 1
